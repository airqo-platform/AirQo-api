1. Get the View API service address by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Values.api.namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Values.api.name }}-svc)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Values.api.namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch its status by running 'kubectl get --namespace {{ .Values.api.namespace }} svc -w {{ .Values.api.name }}-svc'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Values.api.namespace }} {{ .Values.api.name }}-svc --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Values.api.namespace }} -l "app.kubernetes.io/name={{ include "airqo-view-api.name" . }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl port-forward --namespace {{ .Values.api.namespace }} $POD_NAME 8080:{{ .Values.service.targetPort }}
{{- end }}
